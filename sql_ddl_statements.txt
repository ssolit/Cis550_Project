CREATE TABLE Salary
(
    idx                     int,
    company                 varchar(64),
    level                   varchar(20),
    title                   varchar(64),
    totalyearlycompensation int,
    yearsofexperience       float4,
    yearsatcompany          float4,
    basesalary              int,
    stockgrantvalue         int,
    bonus                   int,
    year                    int,
    city                    varchar(30),
    state                   varchar(30),
    country                 varchar(30),
    PRIMARY KEY (idx)
);
